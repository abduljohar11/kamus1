Integrasi komprehensif menyatukan API layer, custom hooks, penanganan form validasi asinkronus backend, hingga aksi manipulasi baris tabel dalam satu modul aplikasi.

ARSITEKTUR RE-USE FORM UNTUK CREATE & UPDATE

Praktik terbaik (*best practice*) dalam aplikasi berskala besar adalah menggunakan **satu komponen Form yang sama** untuk aksi Tambah (Create) dan Edit (Update). Perbedaannya dikontrol melalui status pengiriman objek data aktif (selectedGuru). Jika objek aktif terisi, maka tombol beralih fungsi menjadi PUT/PATCH request menuju database.

01 API Service Lengkap (`src/api/userApi.ts`)

Pastikan seluruh metode HTTP (GET, POST, PUT, DELETE) terpetakan dengan kontrak parameter type data yang jelas. 

import axiosInstance from './axiosInstance';
import { User, ApiResponse } from '../types/user';

export const userApi = {
  // READ
  getAllUsers: async (role: string = ''): Promise<ApiResponse<User[]>> => {
    const response = await axiosInstance.get<ApiResponse<User[]>>(`/api/users?role=${role}`);
    return response.data;
  },

  // CREATE
  createUser: async (payload: Omit<User, 'uid'>): Promise<ApiResponse<User>> => {
    const response = await axiosInstance.post<ApiResponse<User>>('/api/users', payload);
    return response.data;
  },

  // UPDATE
  updateUser: async (uid: string, payload: Partial<User>): Promise<ApiResponse<User>> => {
    const response = await axiosInstance.put<ApiResponse<User>>(`/api/users/${uid}`, payload);
    return response.data;
  },

  // DELETE
  deleteUser: async (uid: string): Promise<ApiResponse<null>> => {
    const response = await axiosInstance.delete<ApiResponse<null>>(`/api/users/${uid}`);
    return response.data;
  }
};

02 Orchestrator Logic Layer (`src/pages/guru/useGuruLogic.ts`)

Custom hook di bawah mengontrol state data daftar guru, melacak objek mana yang sedang diedit, serta menampung pemicu buka-tutup jendela dialog modal. 

import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { useState } from 'react';
import { userApi } from '../../api/userApi';
import { User } from '../../types/user';

export function useGuruLogic() {
  const queryClient = useQueryClient();
  const [selectedGuru, setSelectedGuru] = useState<User | null>(null);
  const [isModalOpen, setIsModalOpen] = useState<boolean>(false);

  // Read Query
  const { data, isLoading } = useQuery({
    queryKey: ['users', 'guru'],
    queryFn: () => userApi.getAllUsers('guru'),
  });

  // Create Mutation
  const createMutation = useMutation({
    mutationFn: userApi.createUser,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['users', 'guru'] });
      closeModal();
    }
  });

  // Update Mutation
  const updateMutation = useMutation({
    mutationFn: ({ uid, payload }: { uid: string; payload: Partial<User> }) => 
      userApi.updateUser(uid, payload),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['users', 'guru'] });
      closeModal();
    }
  });

  // Delete Mutation
  const deleteMutation = useMutation({
    mutationFn: userApi.deleteUser,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['users', 'guru'] });
    }
  });

  const openCreateModal = () => {
    setSelectedGuru(null); // Bersihkan form agar kosong
    setIsModalOpen(true);
  };

  const openEditModal = (guru: User) => {
    setSelectedGuru(guru); // Isi data form dengan rekor baris terpilih
    setIsModalOpen(true);
  };

  const closeModal = () => {
    setSelectedGuru(null);
    setIsModalOpen(false);
  };

  return {
    guruList: data?.data || [],
    isLoading,
    selectedGuru,
    isModalOpen,
    openCreateModal,
    openEditModal,
    closeModal,
    saveGuru: (payload: Omit<User, 'uid'>) => {
      if (selectedGuru) {
        updateMutation.mutate({ uid: selectedGuru.uid, payload });
      } else {
        createMutation.mutate(payload);
      }
    },
    removeGuru: (uid: string) => {
      if (confirm('Apakah Anda yakin ingin menghapus rekor staff ini?')) {
        deleteMutation.mutate(uid);
      }
    }
  };
}

03 Main View & Render Modal Component (`src/pages/guru/GuruList.tsx`)

Komponen utama yang menyatukan komponen tabel aksi tombol pemicu dengan form dialog mengambang (*Popup Form Modal*). 

import React from 'react';
import { useGuruLogic } from './useGuruLogic';
import GuruFormModal from './GuruFormModal';

export default function GuruList(): React.JSX.Element {
  const { 
    guruList, isLoading, isModalOpen, selectedGuru, 
    openCreateModal, openEditModal, closeModal, saveGuru, removeGuru 
  } = useGuruLogic();

  if (isLoading) return <div class="p-6 text-slate-500 text-sm">Sinkronisasi database...</div>;

  return (
    <div class="space-y-6 bg-white p-6 rounded-2xl border border-slate-100 shadow-sm">
      <div class="flex justify-between items-center">
        <h2 class="text-xl font-bold text-slate-900">Daftar Staf Pengajar</h2>
        <button onClick={openCreateModal} class="bg-blue-600 hover:bg-blue-700 text-white text-xs px-4 py-2 rounded-xl font-semibold transition shadow-sm">
          Tambah Guru Baru
        </button>
      </div>

      {/* Komponen Tabel CRUD */}
      <table class="w-full text-left text-sm">
        <thead class="bg-slate-50 text-slate-400 text-xs font-bold uppercase border-b border-slate-100">
          <tr>
            <th class="p-3">Nama Lengkap</th>
            <th class="p-3">NUPTK / Identitas</th>
            <th class="p-3 text-center">Aksi Manajemen</th>
          </tr>
        </thead>
        <tbody class="divide-y divide-slate-100">
          {guruList.map((guru) => (
            <tr key={guru.uid} class="hover:bg-slate-50/50">
              <td class="p-3 font-medium text-slate-900">{guru.nama_lengkap}</td>
              <td class="p-3 font-mono text-xs text-slate-500">{guru.nomor_identitas}</td>
              <td class="p-3 flex justify-center gap-2">
                <button onClick={() => openEditModal(guru)} class="text-xs bg-amber-50 text-amber-600 px-2.5 py-1 rounded-md font-medium hover:bg-amber-100 transition">
                  Edit
                </button>
                <button onClick={() => removeGuru(guru.uid)} class="text-xs bg-red-50 text-red-600 px-2.5 py-1 rounded-md font-medium hover:bg-red-100 transition">
                  Hapus
                </button>
              </td>
            </tr>
          ))}
        </tbody>
      </table>

      {/* Render Jendela Dialog Modal Hanya Jika Status Open Bernilai True */}
      {isModalOpen && (
        <GuruFormModal 
          isOpen={isModalOpen} 
          onClose={closeModal} 
          onSave={saveGuru} 
          currentData={selectedGuru} 
        />
      )}
    </div>
  );
}
