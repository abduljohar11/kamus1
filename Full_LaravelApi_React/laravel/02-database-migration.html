Skema Database: ULID, Enum Role, & Seeders

Cetak biru implementasi otentikasi user berbasis tipe entitas (Guru/Murid) menggunakan struktur data modern terlindungi.

LOGIKA & ALUR DATA

Sistem menggunakan satu tabel master users untuk mempermudah otentikasi tunggal (Sanctum/JWT). Perbedaan hak akses atau fungsionalitas antara Guru dan Murid diisolasi menggunakan data Enum. Untuk keamanan API dan integrasi eksternal, kunci utama dikonfigurasi menggunakan string acak berurutan waktu kronologis (**ULID**) demi mencegah teknik Id Enumeration Attack.

01 Berkas Migrasi Tabel Users

Gunakan metode ulid() sebagai pengganti auto-increment biasa, dan tentukan cakupan data via enum(). 

<?php

use Illuminate\Database\Migrations\Migration;
use Illuminate\Database\Schema\Blueprint;
use Illuminate\Support0\Facades\Schema;

return new class extends Migration
{
    public function up(): void
    {
        Schema::create('users', function (Blueprint $table) {
            // Menggunakan ULID sebagai Primary Key
            $table->ulid('id')->primary(); 
            $table->string('name');
            $table->string('email')->unique();
            $table->string('password');
            
            // Penerapan ENUM untuk klasifikasi role user
            $table->enum('role', ['guru', 'murid'])->default('murid');
            
            // Field opsional penunjang entitas spesifik
            $table->string('nomor_induk')->unique()->comment('NIP untuk Guru, NISN untuk Murid');
            
            $table->rememberToken();
            $table->timestamps();
        });
    }

    public function down(): void
    {
        Schema::dropIfExists('users');
    }
};

02 Konfigurasi Eloquent Model

Agar Eloquent tahu id Anda adalah string ULID dan bukan integer, panggil trait HasUlids. Cast juga kolom password demi enkripsi otomatis. 

<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Concerns\HasUlids;
use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Foundation\Auth\User as Authenticatable;
use Illuminate\Notifications\Notifiable;

class User extends Authenticatable
{
    use HasFactory, Notifiable, HasUlids; // Menambahkan Trait HasUlids

    protected $fillable = [
        'name',
        'email',
        'password',
        'role',
        'nomor_induk',
    ];

    protected $hidden = [
        'password',
        'remember_token',
    ];

    // Casting tipe data data agar aman saat dikelola object orientasi
    protected function casts(): array
    {
        return [
            'email_verified_at' => 'datetime',
            'password' => 'hashed', // Meng-hash otomatis string plain teks password
        ];
    }
}

03 Pengisian Data Awal (UserSeeder)

Sintaksis seeder untuk menyuntikkan data dummy awal saat environment aplikasi dibangun pertama kali. 

<?php

namespace Database\Seeders;

use App\Models\User;
use Illuminate\Database\Seeder;

class UserSeeder extends Seeder
{
    public function run(): void
    {
        // Jalur Logika: Membuat Akun Contoh Perwakilan Guru
        User::create([
            'name'        => 'Budi Setiawan, S.Pd',
            'email'       => 'budi.guru@sekolah.id',
            'password'    => 'password123', // Akan di-hash otomatis oleh property cast model
            'role'        => 'guru',
            'nomor_induk' => '198503242010011003' // Contoh format NIP
        ]);

        // Jalur Logika: Membuat Akun Contoh Perwakilan Murid
        User::create([
            'name'        => 'Ahmad Revan',
            'email'       => 'revan.murid@sekolah.id',
            'password'    => 'password123',
            'role'        => 'murid',
            'nomor_induk' => '0065432190' // Contoh format NISN
        ]);
    }
}
